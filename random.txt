#include<iostream>
using namespace std;

int main() {
	srand(time(NULL));
	int x = rand() % 99;
	int y;
	int z = 0;

	cout << "랜덤한 수: " << x << endl;
	cout << "특정배수 개수: ";
	cin >> y;

	for (int k = 0; k < y; k++) {
		cout << k << "번째 특정 배수: ";
		cin >> z;

		cout << "<특정배수만 출력하는 프로그램>" << endl;

		for (int i = 1; i <= x; i++) {
			if (i % z == 0)
				cout << i << ", ";
			continue;
		}
		cout << endl;

		cout << "<특정배수 제외 출력하는 프로그램>" << endl;

		for (int i = 1; i <= x; i++) {
			if (i % z != 0)
				cout << i << ", ";
			continue;
		}

	}

	return 0;
}
